\section{Définition}
\begin{frame}
	\frametitle{\secname}
	\begin{itemize}
		\item Interrompt le flux normal de traitement
		\item Pour gérer une situation exceptionnelle
		\item Exemple ? \onslide<2->{Mémoire vive épuisée}
		\item Est une forme d’\emph{interruption}
		\item Exemple d’interruption qui n’est pas une exception ? \onslide<3->{Interruption pour basculer à une autre tâche}
		\item Ici : exception au sens génie logiciel (sens Java un peu différent)
	\end{itemize}
\end{frame}

\section{Problème résolu par les exceptions}
\begin{frame}
	\frametitle{\secname}
	\begin{itemize}
		\item Opérations communes souvent faillibles
		\item Exemple : allouer de la mémoire, écrire sur un fichier
		\item Si échec, souhait d’interrompre le flux normal
		\item Éviter de devoir écrire une commande spécifique de traitement à chaque invocation
		\item[⇒] Certaines opérations peuvent lancer une \emph{exception}
	\end{itemize}
\end{frame}

\section{Gestion d’exceptions}
\begin{frame}
	\frametitle{\secname}
	\begin{itemize}
		\item Certains problèmes sont réparables
		\item Exemple : échec lors écriture sur fichier
		\item Le programmeur doit pouvoir intervenir et \emph{gérer} l’exception
		\item Certains problèmes sont difficilement réparables
		\item Exemple : échec lors allocation mémoire
		\item Si pas de gestion spécifique prévue, comportement par défaut
		\item Souvent : passage de l’exception pour traitement possible à un niveau supérieur
		\item Si pas de gestion du tout : arrêt du programme
	\end{itemize}
\end{frame}
